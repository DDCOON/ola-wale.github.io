Here's the method I used to check the Execution Times. It checks the system time in nanoseconds and compares it to the system time after the functions are done running. 

	
#include <cstring>
#include <ctime>
#include <chrono>
auto begin = std::chrono::high_resolution_clock::now();
//measured
auto end = std::chrono::high_resolution_clock::now();
std::cout << "duration \n" << std::chrono::duration_cast<std::chrono::nanoseconds>(end-begin).count() << "ns" << std::endl;
