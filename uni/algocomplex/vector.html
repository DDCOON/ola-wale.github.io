#include <iostream>
#include <cstring>
using namespace std;
struct vec {
	double * arr;
	int capacity;
	int n;
	void increase(int new_size) {
		double* newArr = new double[new_size];
		std::copy(arr, arr + std::min(n, new_size), newArr);
		delete[] arr;
		arr = newArr;
	}
	public:
		// create an empty vector
		vec(): capacity(10), n(0) {
			arr = new double[capacity];
		}
		void push_back(double x) {
			if (n <= capacity) {
				increase(2 * capacity);
				arr[n] = x;
				++n;
			}
		}
		void set(int i, double val) {
			if (i < 0 || i >= n) {
				cout << "range error";
				return;
			}
			arr[i] = val;
		}
		double get(int i) const {
			if (i < 0 || i >= n) {
				cout << "range error";
				return 0;
			}
			return arr[i];
		}

		void printAll() {
			int index = 0;
			//if the array size is less than or equals zero, bail!.
			if (n <= 0) {
				return;
			}
			cout << "\n[";
			while (index < n) {
				cout << arr[index];
				index++;
				if (index == n) {
					cout << "]";
				} else {
					cout << ",";
				}
			}
		}

		void _erase(int index) {
			if (index < 0 || index >= n) {
				cout << "range error";
				return;
			}
			double * newArray = new double[n];
			int newArrayIndex = 0;
			for (int i = 0; i < n; i++) {
				if (i != index) {
					newArray[newArrayIndex] = arr[i];
					newArrayIndex++;
				}
			}
			delete[] arr;
			arr = newArray;
			n--;
		}

	~vec() {
		delete[] arr;
	}


};

int main() {
	vec vec;
	vec.push_back(1);
	vec.push_back(2);
	vec.push_back(3);
	cout << "\nPrint all elements";
	vec.printAll();
